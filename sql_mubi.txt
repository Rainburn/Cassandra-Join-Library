CREATE TABLE list_data (
    user_id int(20),
    list_id int(20),
    list_title VARCHAR(2000),
    list_movie_number int(10),
    list_update_timestamp_utc TIMESTAMP,
    list_creation_timestamp_utc timestamp,
    list_followers int(10),
    list_url VARCHAR(1000),
    list_comments int(10),
    list_description VARCHAR(2000),
    PRIMARY KEY (user_id, list_id)
)

CREATE TABLE movie (
    movie_id int(20),
    movie_title VARCHAR(500),
    movie_release_year int(5),
    movie_url VARCHAR(1000),
    movie_title_language VARCHAR(25),
    movie_popularity int(10),
    movie_image_url VARCHAR(1000),
    director_id int(20),
    director_name VARCHAR(500),
    director_url VARCHAR(1000),
    PRIMARY KEY (movie_id)
)

CREATE TABLE ratings (
    movie_id int(20),
    rating_id int(20),
    rating_url VARCHAR(1000),
    rating_score int(2),
    rating_timestamp_utc timestamp,
    critic VARCHAR(2000),
    critic_likes int(10),
    critic_comments int(10),
    user_id int(20),
    user_trialist int(1),
    PRIMARY KEY (movie_id, rating_id),
    FOREIGN KEY (movie_id) REFERENCES movie(movie_id)
)
